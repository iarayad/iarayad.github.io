Interactive Function Plot

Here’s an interactive visualization of the linear function f(x) = a*x where you can vary the parameter a using the slider below:

import numpy as np
import plotly.graph_objects as go
import plotly.express as px
from plotly.subplots import make_subplots

def f(x, a):
    """Compute f(x) = a*x"""
    return a * x

# Create x values
x = np.linspace(-10, 10, 100)

# Create different values of 'a' for the slider
a_values = np.arange(-3.0, 3.1, 0.5)

# Create the figure
fig = go.Figure()

# Add traces for each value of 'a'
for i, a_val in enumerate(a_values):
    y = f(x, a_val)
    fig.add_trace(
        go.Scatter(
            x=x,
            y=y,
            mode='lines',
            name=f'f(x) = {a_val:.1f}x',
            line=dict(width=3, color='blue'),
            visible=True if i == 30 else False  # Show a=1.0 initially (index 30)
        )
    )

# Create slider steps
steps = []
for i, a_val in enumerate(a_values):
    step = dict(
        method="update",
        args=[{"visible": [False] * len(a_values)},
              {"title": f"Linear Function f(x) = {a_val:.1f}x"}],
        label=f"{a_val:.1f}"
    )
    step["args"][0]["visible"][i] = True  # Toggle i'th trace to "visible"
    steps.append(step)

# Create slider
sliders = [dict(
    active=30,  # Start at a=1.0
    currentvalue={"prefix": "a = "},
    pad={"t": 50},
    steps=steps
)]

# Update layout
fig.update_layout(
    sliders=sliders,
    title="Linear Function f(x) = 1.0x",
    xaxis_title="x",
    yaxis_title="f(x)",
    xaxis=dict(range=[-10, 10], zeroline=True, zerolinewidth=2, zerolinecolor='black'),
    yaxis=dict(range=[-20, 20], zeroline=True, zerolinewidth=2, zerolinecolor='black'),
    showlegend=False,
    width=800,
    height=600,
    template="plotly_white"
)

# Add grid
fig.update_xaxes(showgrid=True, gridwidth=1, gridcolor='lightgray')
fig.update_yaxes(showgrid=True, gridwidth=1, gridcolor='lightgray')

fig.show()
